功能优化要求#

      package com.sky.entity;

import lombok.Data;

import java.math.BigDecimal;
import java.math.RoundingMode;

@Data
public class Location {
    private Double lat;
    private Double lng;

    public void format() {
        this.lat = this.round(this.lat);
        this.lng = this.round(this.lng);
    }

    private Double round(Double value) {
        return BigDecimal.valueOf(value)
                .setScale(6, RoundingMode.HALF_UP)
                .doubleValue();
    }
}

  6. 创建BauduMapUtils工具类

    package com.sky.utils;

import com.alibaba.fastjson.JSON;
import com.sky.entity.DistanceResponse;
import com.sky.entity.Location;
import com.sky.properties.BaiduProoerties;
import com.sky.properties.ShopProperties;
import com.sky.entity.GeoCodeResponse;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.HashMap;
import java.util.Map;


/**
 * 百度地图工具类
 */
@Component
public class BaiduMapUtils {
    //    编码API
    private static final String CODINGAPI = "https://api.map.baidu.com/geocoding/v3/";
    //    路线规划API
    private static final String DISTANCEAPI = "https://api.map.baidu.com/directionlite/v1/riding";

    @Autowired
    private BaiduProoerties baiduProoerties;

    @Autowired
    private ShopProperties shopProperties;

    public GeoCodeResponse getCoding(String address) {
        Map<String,String> map = new HashMap();
        map.put("ak", baiduProoerties.getAk());
        map.put("address", address);
        map.put("output", "json");
        String json = HttpClientUtil.doGet(CODINGAPI, map);
        return JSON.parseObject(json, GeoCodeResponse.class);
    }

    public Double getDistance(Location location) {
        location.format();
        Map<String,String> map = new HashMap();
        map.put("ak", baiduProoerties.getAk());
//        起点：门店
        map.put("origin", shopProperties.getLat() + "," + shopProperties.getLng());
//        终点：用户
        map.put("destination", location.getLat() + "," + location.getLng());
        map.put("steps_info", "0");
        String json = HttpClientUtil.doGet(DISTANCEAPI, map);
        return JSON.parseObject(json, DistanceResponse.class).getResult().getRoutes().get(0).getDistance();
    }
}

  7. 修改用户下单功能

    • ` 
      @Autowired
private BaiduMapUtils baiduMapUtils;

/**
* 用户下单
*
* @param dto
* @return
*/
@Override
@Transactional
public OrderSubmitVO submit(OrdersSubmitDTO dto) {
   Long userId = BaseContext.getCurrentId();
   AddressBook addressBook = addressBookMapper.getById(dto.getAddressBookId());
   if (addressBook == null) {
       throw new AddressBookBusinessException(MessageConstant.ADDRESS_BOOK_IS_NULL);
   }
   String address = addressBook.getProvinceName() + addressBook.getCityName() + addressBook.getDistrictName() + addressBook.getDetail();
//        距离校验
   GeoCodeResponse coding = baiduMapUtils.getCoding(address);
   Location location = coding.getResult().getLocation();
   Double distance = baiduMapUtils.getDistance(location);
   log.info("距离：{}",distance);
   if (distance > 5000) {
       throw new AddressBookBusinessException(MessageConstant.DISTANCE_TOO_FAR);
   }


   User user = userMapper.getById(userId);
   List<ShoppingCart> shoppingCartList = shoppingCartMapper.list(userId);
   if (shoppingCartList == null || shoppingCartList.size() == 0) {
       throw new ShoppingCartBusinessException(MessageConstant.SHOPPING_CART_IS_NULL);
   }

//        构造订单数据
   Orders orders = new Orders();
   BeanUtils.copyProperties(dto, orders);
   orders.setNumber(String.valueOf(System.currentTimeMillis()));
   orders.setStatus(Orders.PENDING_PAYMENT);
   orders.setUserId(userId);
   orders.setOrderTime(LocalDateTime.now());
   orders.setPhone(addressBook.getPhone());
   orders.setAddress(address);
   orders.setUserName(user.getName());
   orders.setConsignee(addressBook.getConsignee());
   orders.setPayStatus(Orders.UN_PAID);
   orderMapper.insert(orders);

   List<OrderDetail> orderDetailList = new ArrayList<>();
   for (ShoppingCart shoppingCart : shoppingCartList) {
       OrderDetail orderDetail = new OrderDetail();
       BeanUtils.copyProperties(shoppingCart, orderDetail);
       orderDetail.setOrderId(orders.getId());
       orderDetailList.add(orderDetail);
   }
   orderDetailMapper.insertBatch(orderDetailList);

   OrderSubmitVO orderSubmitVO = OrderSubmitVO.builder()
           .id(orders.getId())
           .orderNumber(orders.getNumber())
           .orderAmount(orders.getAmount())
           .orderTime(orders.getOrderTime())
           .build();
   return orderSubmitVO;
}
